import sys 
from PySide6.QtCore import * 
from PySide6.QtWidgets import * 
from PySide6.QtGui import * 
from PySide6.QtMultimedia import QSoundEffect  

class frog_area(QWidget):  
    
    def __init__(self): 
        QWidget.__init__(self,None)

        self.frame_no = 0 
        self.images = [QPixmap ("py_toggle\frogs\frog_"+ str(i+1)+".png") 
                       for i in range(5)]  

        self.timer = QTimer(self)  
        self.timer.timeout.connect(self.update_value) 
        self.timer.start(75) 
        # self.QSE = QSoundEffect()
        # self.QSE.setSource() 

    def paintEvent(self, e): 
      p=QPainter() 
      p.begin(self) 
      p.drawPixmap(QRect(0,0,320,320), self.images[self.frame_no]) 
      p.end()  

    def update_value(self): 
      self.frame_no+=1 
      if self.frame_no >= 5 :
        self.frame_no = 0 
        # self.QSE.play() 

      self.update() 

class Simple_animation_window(QWidget): 
  def __init__(self): 
    QWidget.__init__(self, None) 

    self.while_run = True 
    self.anim_area = frog_area() 
    layout = QVBoxLayout() 
    layout.addWidget(self.anim_area) 

    btn = QPushButton("Pause",self)
    layout.addWidget(btn) 
    btn.clicked.connect(self.push_pause)
    
     
    self.setLayout(layout) 
    self.setMinimumSize(330,400) 

  def push_pause(self): 
    if self.while_run == True: 
      self.anim_area.timer.stop()
      self.sender().setText("Play") 
      self.while_run = False
    
    else : 
      self.while_run = True
      self.anim_area.timer.start()
      self.sender().setText("Pause")



def main(): 
  app = QApplication(sys.argv) 

  w=Simple_animation_window() 
  w.show() 
  return app.exec() 

if __name__ == "__main__" : 
  sys.exit(main())
